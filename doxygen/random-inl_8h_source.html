69template <typename T, typename Generator>
70[[nodiscard]] T generateScalarUniform(const T& min, const T& max, Generator& generator) {
71 // Real types
72 if constexpr (std::is_floating_point_v<T>) {
73 UniformRealDist<T> dist(min, max);
74 return dist(generator);
75 }
76 // Integer types
77 else if constexpr (is_compatible_to_uniform_int_distribution_v<T>) {
78 UniformIntDist<T> dist(min, max);
79 return dist(generator);
80 }
81}

83// Generates a random vector/matrix element-wisely
84template <typename Derived, typename Generator>
85[[nodiscard]] typename Derived::PlainObject generateMatrixUniform(
86 const Eigen::MatrixBase<Derived>& min,
87 const Eigen::MatrixBase<Derived>& max,
88 Generator& generator) {
89 // Dynamic matrix
90 if constexpr (!Derived::IsVectorAtCompileTime && Derived::SizeAtCompileTime == Eigen::Dynamic) {
91 return Derived::PlainObject::NullaryExpr(min.rows(), min.cols(), [&](const int i, const int j) {
92 return generateScalarUniform<typename Derived::Scalar>(min(i, j), max(i, j), generator);
93 });
94 }
95 // Fixed matrix
96 else if constexpr (
97 !Derived::IsVectorAtCompileTime && Derived::SizeAtCompileTime != Eigen::Dynamic) {
98 return Derived::PlainObject::NullaryExpr([&](const int i, const int j) {
99 return generateScalarUniform<typename Derived::Scalar>(min(i, j), max(i, j), generator);
100 });
101 }
102 // Dynamic vector
103 else if constexpr (
104 Derived::IsVectorAtCompileTime && Derived::SizeAtCompileTime == Eigen::Dynamic) {
105 return Derived::PlainObject::NullaryExpr(min.size(), [&](const int i) {
106 return generateScalarUniform<typename Derived::Scalar>(min[i], max[i], generator);
107 });
108 }
109 // Fixed vector
110 else if constexpr (
111 Derived::IsVectorAtCompileTime && Derived::SizeAtCompileTime != Eigen::Dynamic) {
112 return Derived::PlainObject::NullaryExpr([&](const int i) {
113 return generateScalarUniform<typename Derived::Scalar>(min[i], max[i], generator);
114 });
115 }

130template <typename T, typename Generator>
131[[nodiscard]] T generateScalarNormal(const T& mean, const T& sigma, Generator& generator) {
132 // Real types
133 if constexpr (std::is_floating_point_v<T>) {
134 NormalRealDist<T> dist(mean, sigma);
135 return dist(generator);
136 }
137 // Integer types
138 else if constexpr (is_compatible_to_uniform_int_distribution_v<T>) {
139 const float realNumber = NormalRealDist<float>(mean, sigma)(generator);
140 return std::round(realNumber);
141 }
142}
