35enum class VertexConstraintType {
36 Position, // Target the vertex position
37 Plane, // point-to-plane distance using the target normal
38 Normal, // point-to-plane distance using the source (body) normal
39 SymmetricNormal, // Point-to-plane using a 50/50 mix of source and target normal
40};
