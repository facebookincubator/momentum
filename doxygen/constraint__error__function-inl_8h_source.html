30template <typename T, class Data, size_t FuncDim, size_t NumVec, size_t NumPos>
31double ConstraintErrorFunctionT<T, Data, FuncDim, NumVec, NumPos>::getError(
32 const ModelParametersT<T>& /* params */,
33 const SkeletonStateT<T>& state) {
34 MT_PROFILE_FUNCTION();
35
36 // loop over all constraints and calculate the error
37 FuncType f;
38 double error = 0.0;
39 for (size_t iConstr = 0; iConstr < constraints_.size(); ++iConstr) {
40 const Data& constr = constraints_[iConstr];
41 if (constr.weight != 0) {
42 evalFunction(iConstr, state.jointState.at(constr.parent), f);
43 error += constr.weight * loss_.value(f.squaredNorm());
44 }
45 }
46 return this->weight_ * error;
47}

49template <typename T, class Data, size_t FuncDim, size_t NumVec, size_t NumPos>
50double ConstraintErrorFunctionT<T, Data, FuncDim, NumVec, NumPos>::getJacobianForSingleConstraint(
51 const JointStateListT<T>& jointStates,
52 const size_t iConstr,
53 Ref<Eigen::MatrixX<T>> jacobian,
54 Ref<Eigen::VectorX<T>> residual) {
55 const Data& constr = this->constraints_[iConstr];
56 if (constr.weight == 0) {
57 return 0;
58 }
59
60 FuncType f;
61 std::array<VType, NumVec> v;
62 std::array<DfdvType, NumVec> dfdv;
63 evalFunction(iConstr, jointStates.at(constr.parent), f, v, dfdv);
64 const T sqrError = f.squaredNorm();
65 const T w = constr.weight * this->weight_;
66 const double error = w * this->loss_.value(sqrError);
67 const T deriv = std::sqrt(w * this->loss_.deriv(sqrError));
68
69 // The input jacobian is a subblock from the full jacobian tailored for this error function, so
70 // the rows start from zero.
71 const size_t rowIndex = FuncDim * iConstr;
72 // The gradient of the loss function needs to be split between J and r, so both J'J and
73 // J'r are scaled correctly. The factor 2 from the square term is accounted for in the base
74 // class function.
75 residual.template middleRows<FuncDim>(rowIndex).noalias() = deriv * f;
76
77 // Small optimization for early termination
78 if (isApprox<T>(deriv, T(0), Eps<T>(1e-9, 1e-16))) {
79 return error;
80 }
81 const bool zeroDeriv =
82 !std::any_of(dfdv.begin(), dfdv.end(), [](const auto& vec) { return !vec.isZero(); });
83 if (zeroDeriv) {
84 return error;
85 }
86
87 // shorthands for code readability
88 const auto outerIndexPtr = this->parameterTransform_.transform.outerIndexPtr();
89 const auto innerIndexPtr = this->parameterTransform_.transform.innerIndexPtr();
90 const auto valuePtr = this->parameterTransform_.transform.valuePtr();
91 auto&& jac = jacobian.template middleRows<FuncDim>(rowIndex);
92
93 // Fill in the jacobian by walking up the joint hierarchy for each constraint.
94 // Need to consider both active joints and enabled model parameters.
95 size_t jntIndex = constr.parent;
96 while (jntIndex != kInvalidIndex) {
97 // check for valid index
98 MT_CHECK(jntIndex < this->skeleton_.joints.size());
99
100 const auto& jntState = jointStates[jntIndex];
101 const size_t jntParamIndex = jntIndex * kParametersPerJoint;
102
103 for (size_t jVec = 0; jVec < NumVec; ++jVec) {
104 if (dfdv[jVec].isZero()) {
105 continue;
106 }
107
108 Eigen::Vector3<T> offset;
109 if (jVec < NumPos) {
110 offset.noalias() = v[jVec] - jntState.translation();
111 } else {
112 offset = v[jVec];
113 }
114
115 // Translational dofs -- only affect POINT
116 if (jVec < NumPos) {
117 for (size_t d = 0; d < 3; ++d) {
118 if (!this->activeJointParams_[jntParamIndex + d]) {
119 continue;
120 }
121
122 const FuncType jc = deriv * dfdv[jVec] * jntState.getTranslationDerivative(d);
123 for (auto index = outerIndexPtr[jntParamIndex + d];
124 index < outerIndexPtr[jntParamIndex + d + 1];
125 ++index) {
126 if (this->enabledParameters_.test(innerIndexPtr[index])) {
127 jac.col(innerIndexPtr[index]).noalias() += jc * valuePtr[index];
128 }
129 }
130 }
131 }
132
133 // Rotational dofs -- affect both POINT and AXIS
134 for (size_t d = 0; d < 3; ++d) {
135 if (!this->activeJointParams_[jntParamIndex + 3 + d]) {
136 continue;
137 }
138
139 const FuncType jc = deriv * dfdv[jVec] * jntState.getRotationDerivative(d, offset);
140 for (auto index = outerIndexPtr[jntParamIndex + 3 + d];
141 index < outerIndexPtr[jntParamIndex + 3 + d + 1];
142 ++index) {
143 if (this->enabledParameters_.test(innerIndexPtr[index])) {
144 jac.col(innerIndexPtr[index]).noalias() += jc * valuePtr[index];
145 }
146 }
147 }
148
149 // Scale dof -- only affect POINT
150 if (jVec < NumPos) {
151 if (this->activeJointParams_[jntParamIndex + 6]) {
152 const FuncType jc = deriv * dfdv[jVec] * jntState.getScaleDerivative(offset);
153 for (auto index = outerIndexPtr[jntParamIndex + 6];
154 index < outerIndexPtr[jntParamIndex + 6 + 1];
155 ++index) {
156 if (this->enabledParameters_.test(innerIndexPtr[index])) {
157 jac.col(innerIndexPtr[index]).noalias() += jc * valuePtr[index];
158 }
159 }
160 }
161 }
162 }
163 // go up the hierarchy to the parent
164 jntIndex = this->skeleton_.joints[jntIndex].parent;
165 }
166
167 return error;
168}

170template <typename T, class Data, size_t FuncDim, size_t NumVec, size_t NumPos>
171double ConstraintErrorFunctionT<T, Data, FuncDim, NumVec, NumPos>::getJacobian(
172 const ModelParametersT<T>& /*params*/,
173 const SkeletonStateT<T>& state,
174 Ref<Eigen::MatrixX<T>> jacobian,
175 Ref<Eigen::VectorX<T>> residual,
176 int& usedRows) {
177 MT_PROFILE_FUNCTION();
178 usedRows = getJacobianSize();
179
180 double error = 0.0;
181 for (size_t iConstr = 0; iConstr < this->constraints_.size(); ++iConstr) {
182 error += getJacobianForSingleConstraint(state.jointState, iConstr, jacobian, residual);
183 }
184 return error;

187template <typename T, class Data, size_t FuncDim, size_t NumVec, size_t NumPos>
188double ConstraintErrorFunctionT<T, Data, FuncDim, NumVec, NumPos>::getGradientForSingleConstraint(
189 const JointStateListT<T>& jointStates,
190 size_t iConstr,
191 Ref<VectorX<T>> /*gradient*/) {
192 const Data& constr = this->constraints_[iConstr];
193 if (constr.weight == 0) {
194 return 0;
195 }
196
197 FuncType f;
198 std::array<VType, NumVec> v;
199 std::array<DfdvType, NumVec> dfdv;
200 evalFunction(iConstr, jointStates.at(constr.parent), f, v, dfdv);
201 if (f.isZero()) {
202 return 0;
203 }
204
205 const T sqrError = f.squaredNorm();
206 const T w = constr.weight * this->weight_;
207 const double error = w * this->loss_.value(sqrError);
208 // The gradient is scaled by the gradient of the loss function; factor 2 falls out of the
209 // square.
210 const FuncType deriv = T(2) * w * this->loss_.deriv(sqrError) * f;
211
212 const auto outerIndexPtr = this->parameterTransform_.transform.outerIndexPtr();
213 const auto innerIndexPtr = this->parameterTransform_.transform.innerIndexPtr();
214 const auto valuePtr = this->parameterTransform_.transform.valuePtr();
215 size_t jntIndex = constr.parent;
216 while (jntIndex != kInvalidIndex) {
217 // check for valid index
218 MT_CHECK(jntIndex < this->skeleton_.joints.size());
219
220 const auto& jntState = jointStates[jntIndex];
221 const size_t jntParamIndex = jntIndex * kParametersPerJoint;
222
223 for (size_t jVec = 0; jVec < NumVec; ++jVec) {
224 if (dfdv[jVec].isZero()) {
225 continue;
226 }
227
228 Eigen::Vector3<T> offset;
229 if (jVec < NumPos) {
230 offset.noalias() = v[jVec] - jntState.translation();
231 } else {
232 offset = v[jVec];
233 }
234
235 // Translational dofs -- only affect POINT
236 if (jVec < NumPos) {
237 for (size_t d = 0; d < 3; d++) {
238 if (!this->activeJointParams_[jntParamIndex + d]) {
239 continue;
240 }
241
242 // joint space gradient
243 const T val = deriv.dot(dfdv[jVec] * jntState.getTranslationDerivative(d));
244 // multiply with parameter transform to get model parameter gradient
245 for (auto index = outerIndexPtr[jntParamIndex + d];
246 index < outerIndexPtr[jntParamIndex + d + 1];
247 ++index) {
248 if (this->enabledParameters_.test(innerIndexPtr[index])) {
249 jointGrad_[innerIndexPtr[index]] += val * valuePtr[index];
250 }
251 }
252 }
253 }
254
255 // Rotational dofs -- affect both POINT and AXIS
256 for (size_t d = 0; d < 3; d++) {
257 if (!this->activeJointParams_[jntParamIndex + 3 + d]) {
258 continue;
259 }
260
261 // joint space gradient
262 const T val = deriv.dot(dfdv[jVec] * jntState.getRotationDerivative(d, offset));
263 // multiply with parameter transform to get model parameter gradient
264 for (auto index = outerIndexPtr[jntParamIndex + 3 + d];
265 index < outerIndexPtr[jntParamIndex + 3 + d + 1];
266 ++index) {
267 if (this->enabledParameters_.test(innerIndexPtr[index])) {
268 jointGrad_[innerIndexPtr[index]] += val * valuePtr[index];
269 }
270 }
271 }
272
273 // Scale dof -- only affect POINT
274 if (jVec < NumPos) {
275 if (this->activeJointParams_[jntParamIndex + 6]) {
276 // joint space gradient
277 const T val = deriv.dot(dfdv[jVec] * jntState.getScaleDerivative(offset));
278 // multiply with parameter transform to get model parameter gradient
279 for (auto index = outerIndexPtr[jntParamIndex + 6];
280 index < outerIndexPtr[jntParamIndex + 6 + 1];
281 ++index) {
282 if (this->enabledParameters_.test(innerIndexPtr[index])) {
283 jointGrad_[innerIndexPtr[index]] += val * valuePtr[index];
284 }
285 }
286 }
287 }
288 }
289 // go up the skeleton hierarchy to the parent joint
290 jntIndex = this->skeleton_.joints[jntIndex].parent;
291 }
292
293 return error;
294}

296template <typename T, class Data, size_t FuncDim, size_t NumVec, size_t NumPos>
297double ConstraintErrorFunctionT<T, Data, FuncDim, NumVec, NumPos>::getGradient(
298 const ModelParametersT<T>& /*unused*/,
299 const SkeletonStateT<T>& state,
300 Ref<Eigen::VectorX<T>> gradient) {
301 MT_PROFILE_FUNCTION();
302
303 // initialize joint gradients storage
304 jointGrad_.setZero();
305
306 double error = 0.0;
307 for (size_t iConstr = 0; iConstr < this->constraints_.size(); ++iConstr) {
308 error += getGradientForSingleConstraint(state.jointState, iConstr, gradient);
309 }
310
311 gradient += jointGrad_;
312 return error;
313}
