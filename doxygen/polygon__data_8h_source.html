20// Structure for holding arbitrary polygon data. We will read the data as-is from the
21// FBX file and then triangulate at the end for momentum. Doing it this way allows us
22// to split up the functionality (triangulating while you read complicates handling
23// texture coordinates for example).
24struct PolygonData {
25 // We will pack polygons in an array like this:
26 // The indices array will contain all the polygons indices packed back-to-back.
27 // 1 2 3 4 2 3 4 6 ...
28 // The offsets array will include an offset for the start and end of each polygon.
29 // 0 4 8 ...
30 // The advantage of doing it this way is that we have random access into the
31 // polygons. The disadvantage (compared to just providing an array of vertex counts)
32 // is that the offset array should always have one more than the number of polygons,
33 // which is a little confusing.
34 std::vector<uint32_t> indices;
35 std::vector<uint32_t> offsets;
36
37 // The texture coordinate indices for each face.
38 // Uses the same offsets array as the indices array.
39 // Note that this is allowed to be empty, but if not empty it
40 // must be the same size as the indices array.
41 std::vector<uint32_t> textureIndices;
42
43 PolygonData() : offsets{0} {}
44
45 // If there's a problem with the mesh, returns an error message.
46 // If everything is kosher, returns an empty string.
47 // This function is used in a few places when reading the mesh from
48 // disk to make sure it's valid.
49 [[nodiscard]] std::string errorMessage(size_t numVertices) const;
50
51 // Similar to errorMessage, but the issue is not severe enough to terminate.
52 [[nodiscard]] std::string warnMessage(size_t numTexVertices) const;
53
54 size_t numPolygons() const;
55};
